176float3 estimateSurfaceNormal(float3 position, float time)
177{
178 const float3 xOffset = (float3)(SURFACE_NORMAL_EPSILON, 0, 0);
179 const float3 yOffset = (float3)(0, SURFACE_NORMAL_EPSILON, 0);
180 const float3 zOffset = (float3)(0, 0, SURFACE_NORMAL_EPSILON);
181
182 float x = signedDistanceEstimation(position + xOffset, time).w - signedDistanceEstimation(position - xOffset, time).w;
183 float y = signedDistanceEstimation(position + yOffset, time).w - signedDistanceEstimation(position - yOffset, time).w;
184 float z = signedDistanceEstimation(position + zOffset, time).w - signedDistanceEstimation(position - zOffset, time).w;
185
186 return normalize((float3)(x, y, z));
187}

196float3 trace(Ray ray, float time)
197{
198 float totalDistance;
199 int steps;
200 for (steps = 0, totalDistance = 0; steps < MAXIMUM_MARCH_STEPS && totalDistance < MAXIMUM_MARCH_DISTANCE; steps++)
201 {
202 float3 currentPosition = ray.position + (ray.direction * totalDistance);
203 float4 colourAndDistance = signedDistanceEstimation(currentPosition, time);
204 totalDistance += colourAndDistance.w;
205
206 // Hit the surface of an object
207 if (colourAndDistance.w <= SURFACE_INTERSECTION_EPSILON)
208 {
209 float3 new_colour = colourAndDistance.xyz;
210
211 float3 normal = estimateSurfaceNormal(currentPosition, time);
212 float percent = (float)steps / (float)MAXIMUM_MARCH_STEPS;
213
214 // Render normals
215 new_colour = (normal + (float3)(1)) * 0.5f;
216
217 // Non-shaded
218 new_colour = new_colour * (1 - percent);
219
220 // Phong
221 //colour = Vector3::multiplyComponents(colour , phong(normal, direction));
222
223 return new_colour;
224 }
225 }
226
227 return (float3)(0);
228}

238Ray getCameraRay(float2 screen_coordinate, float3 camera_position, float3 camera_facing, float aspect_ratio)
239{
240 const float theta = CAMERA_VERTICAL_FOV_DEGREES * PI / 180.0f;
241 const float h = tan(theta / 2);
242 const float viewport_height = 2.0f * h;
243 const float viewport_width = aspect_ratio * viewport_height;
244
245 float3 w = normalise(camera_facing);
246 float3 u = normalise(crossProduct(CAMERA_UP_AXIS, w));
247 float3 v = crossProduct(w, u);
248
249 float3 horizontal = u * CAMERA_FOCUS_DISTANCE * viewport_width;
250 float3 vertical = v * CAMERA_FOCUS_DISTANCE * viewport_height;
251 float3 screenLowerLeftCorner = camera_position - horizontal / 2 - vertical / 2 - w * CAMERA_FOCUS_DISTANCE;
252
253 float3 screenPosition = screenLowerLeftCorner + horizontal * screen_coordinate.x + vertical * screen_coordinate.y - camera_position;
254
255 Ray r;
256 r.position = camera_position + screenPosition;
257 r.direction = normalise(screenPosition);
258 return r;
259}
