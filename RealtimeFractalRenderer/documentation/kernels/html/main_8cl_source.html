125float3 estimateSurfaceNormal(float3 position, float time)
126{
127 const float3 xOffset = (float3)(SURFACE_NORMAL_EPSILON, 0, 0);
128 const float3 yOffset = (float3)(0, SURFACE_NORMAL_EPSILON, 0);
129 const float3 zOffset = (float3)(0, 0, SURFACE_NORMAL_EPSILON);
130
131 float x = signedDistanceEstimation(position + xOffset, time).w - signedDistanceEstimation(position - xOffset, time).w;
132 float y = signedDistanceEstimation(position + yOffset, time).w - signedDistanceEstimation(position - yOffset, time).w;
133 float z = signedDistanceEstimation(position + zOffset, time).w - signedDistanceEstimation(position - zOffset, time).w;
134
135 return normalize((float3)(x, y, z));
136}

145float3 trace(float3 position, float3 direction, float time)
146{
147 float totalDistance;
148 int steps;
149 for (steps = 0, totalDistance = 0; steps < MAXIMUM_MARCH_STEPS && totalDistance < MAXIMUM_MARCH_DISTANCE; steps++)
150 {
151 float3 currentPosition = position + (float3)(direction.x * totalDistance, direction.y * totalDistance, direction.z * totalDistance);
152 float4 colourAndDistance = signedDistanceEstimation(currentPosition, time);
153 totalDistance += colourAndDistance.w;
154
155 // Hit the surface of an object
156 if (colourAndDistance.w <= SURFACE_INTERSECTION_EPSILON)
157 {
158 float3 new_colour = colourAndDistance.xyz;
159
160 float3 normal = estimateSurfaceNormal(currentPosition, time);
161 float percent = (float)steps / (float)MAXIMUM_MARCH_STEPS;
162
163 // Render normals
164 new_colour = (normal + (float3)(1)) * 0.5f;
165
166 // Non-shaded
167 new_colour = new_colour * (1 - percent);
168
169 // Phong
170 //colour = Vector3::multiplyComponents(colour , phong(normal, direction));
171
172 return new_colour;
173 }
174 }
175
176 return (float3)(0);
177}

200Ray getCameraRay(float2 screen_coordinate, float3 camera_position, float3 camera_facing, float3 camera_up,
201 float vertical_fov_degrees, float aspect_ratio, float focus_distance)
202{
203 float theta = degreesToRadians(vertical_fov_degrees);
204 float h = tan(theta / 2);
205 float viewport_height = 2.0f * h;
206 float viewport_width = aspect_ratio * viewport_height;
207
208 float3 w = normalise(camera_facing);
209 float3 u = normalise(crossProduct(camera_up, w));
210 float3 v = crossProduct(w, u);
211
212 float3 horizontal = u * focus_distance * viewport_width;
213 float3 vertical = v * focus_distance * viewport_height;
214 float3 screenLowerLeftCorner = camera_position - horizontal / 2 - vertical / 2 - w * focus_distance;
215
216 float3 screenPosition = screenLowerLeftCorner + horizontal * screen_coordinate.x + vertical * screen_coordinate.y - camera_position;
217
218 Ray r;
219 r.position = camera_position + screenPosition;
220 r.direction = normalise(screenPosition);
221 return r;
222}
