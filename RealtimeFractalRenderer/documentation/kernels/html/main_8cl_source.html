123float3 estimateSurfaceNormal(float3 surface, float time)
124{
125 float3 xOffset = (float3)(SURFACE_NORMAL_EPSILON, 0, 0);
126 float3 yOffset = (float3)(0, SURFACE_NORMAL_EPSILON, 0);
127 float3 zOffset = (float3)(0, 0, SURFACE_NORMAL_EPSILON);
128
129 float x = signedDistanceEstimation(surface + xOffset, time).w - signedDistanceEstimation(surface - xOffset, time).w;
130 float y = signedDistanceEstimation(surface + yOffset, time).w - signedDistanceEstimation(surface - yOffset, time).w;
131 float z = signedDistanceEstimation(surface + zOffset, time).w - signedDistanceEstimation(surface - zOffset, time).w;
132
133 return normalize((float3)(x, y, z));
134}

141float3 trace(float3 pos, float3 dir, float time)
142{
143 float totalDistance;
144 int steps;
145 for (steps = 0, totalDistance = 0; steps < MAXIMUM_MARCH_STEPS && totalDistance < MAXIMUM_MARCH_DISTANCE; steps++)
146 {
147 float3 currentPosition = pos + (float3)(dir.x * totalDistance, dir.y * totalDistance, dir.z * totalDistance);
148 float4 colourAndDistance = signedDistanceEstimation(currentPosition, time);
149 totalDistance += colourAndDistance.w;
150
151 // Hit the surface of an object
152 if (colourAndDistance.w <= SURFACE_INTERSECTION_EPSILON)
153 {
154 float3 new_colour = colourAndDistance.xyz;
155
156 float3 normal = estimateSurfaceNormal(currentPosition, time);
157 float percent = (float)steps / (float)MAXIMUM_MARCH_STEPS;
158
159 // Render normals
160 new_colour = (normal + (float3)(1)) * 0.5f;
161
162 // Non-shaded
163 new_colour = new_colour * (1 - percent);
164
165 // Phong
166 //colour = Vector3::multiplyComponents(colour , phong(normal, direction));
167
168 return new_colour;
169 }
170 }
171
172 return (float3)(0);
173}

193Ray getCameraRay(float2 screen_coordinate, float3 camera_position, float3 camera_facing, float3 camera_up,
194 float vertical_fov_degrees, float aspect_ratio, float focus_distance)
195{
196 float theta = degreesToRadians(vertical_fov_degrees);
197 float h = tan(theta / 2);
198 float viewport_height = 2.0 * h;
199 float viewport_width = aspect_ratio * viewport_height;
200
201 float3 w = normalise(camera_facing);
202 float3 u = normalise(crossProduct(camera_up, w));
203 float3 v = crossProduct(w, u);
204
205 float3 horizontal = u * focus_distance * viewport_width;
206 float3 vertical = v * focus_distance * viewport_height;
207 float3 screenLowerLeftCorner = camera_position - horizontal / 2 - vertical / 2 - w * focus_distance;
208
209 float3 screenPosition = screenLowerLeftCorner + horizontal * screen_coordinate.x + vertical * screen_coordinate.y - camera_position;
210
211 Ray r;
212 r.position = camera_position + screenPosition;
213 r.direction = normalise(screenPosition);
214 return r;
215}
