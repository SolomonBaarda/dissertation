61float3 estimateSurfaceNormal(const float3 position, const float time)
62{
63 const float3 xOffset = (float3)(SURFACE_NORMAL_EPSILON, 0, 0);
64 const float3 yOffset = (float3)(0, SURFACE_NORMAL_EPSILON, 0);
65 const float3 zOffset = (float3)(0, 0, SURFACE_NORMAL_EPSILON);
66
67 float x = signedDistanceEstimation(position + xOffset, time) - signedDistanceEstimation(position - xOffset, time);
68 float y = signedDistanceEstimation(position + yOffset, time) - signedDistanceEstimation(position - yOffset, time);
69 float z = signedDistanceEstimation(position + zOffset, time) - signedDistanceEstimation(position - zOffset, time);
70
71 return normalize((float3)(x, y, z));

80float calculateSoftShadow(const float3 pointOnGeometry, const float time, const float3 lightPosition)
81{
82 // https://www.iquilezles.org/www/articles/rmshadows/rmshadows.htm
83
84 float shadow = 1.0f;
85 float ph = 1e20;
86
87 float3 distanceVector = lightPosition - pointOnGeometry;
88 float3 directionFromGeometryToLight = normalise(distanceVector);
89 float distanceFromGeometryToLight = magnitude(distanceVector);
90
91 for (float totalDistance = SURFACE_SHADOW_EPSILON; totalDistance < distanceFromGeometryToLight; )
92 {
93 float3 currentPosition = pointOnGeometry + directionFromGeometryToLight * totalDistance;
94 float distance = signedDistanceEstimation(currentPosition, time);
95
96 // Hit the surface of an object
97 // Therefore the original point must be in shadow
98 if (distance <= SURFACE_INTERSECTION_EPSILON)
99 {
100 return 0.0f;
101 }
102
103 // Calculate soft shadow values
104 float y = distance * distance / (2.0f * ph);
105 float d = sqrt(distance * distance - y * y);
106 shadow = min(shadow, SURFACE_SHADOW_FALLOFF * d / max(0.0f, totalDistance - y));
107 ph = distance;
108
109 totalDistance += distance;
110 }
111
112 return shadow;
113}

158float3 trace(const Ray ray, const float time)
159{
160 float totalDistance = 0.0f;
161 float closestDistanceToGeometry = FLOAT_MAX_VALUE;
162 float3 currentPosition;
163 int steps = 0;
164 for (; steps < MAXIMUM_MARCH_STEPS && totalDistance < MAXIMUM_MARCH_DISTANCE; steps++)
165 {
166 currentPosition = ray.position + (ray.direction * totalDistance);
167 float distance = signedDistanceEstimation(currentPosition, time);
168 totalDistance += distance;
169
170 // Record the closest distance to the geometry
171 if (distance < closestDistanceToGeometry)
172 {
173 closestDistanceToGeometry = distance;
174 }
175
176 // Hit the surface of an object
177#if INCREASE_INTERSECTION_EPSILON_LINEARLY
178 if (distance <= SURFACE_INTERSECTION_EPSILON * totalDistance)
179#else
180 if (distance <= SURFACE_INTERSECTION_EPSILON)
181#endif
182 {
183 Material material = getMaterial(currentPosition, time);
184 Light light = getLight(time);
185
186 float3 ambient = material.ambient;
187 float3 diffuse = material.diffuse;
188 float3 specular = material.specular;
189 float shadow = 1.0f;
190
191 const float3 normal = estimateSurfaceNormal(currentPosition, time);
192 const float3 lightDirection = normalise(light.position - currentPosition);
193
194#if DO_RENDER_SURFACE_NORMALS
195 // Set ambient and diffuse colours to be surface normal
196 ambient = (normal + (float3)(1.0f)) * 0.5f;
197 diffuse = ambient;
198#endif
199
200 // Ambient
201#if DO_AMBIENT_LIGHTING
202 ambient *= light.ambient;
203#endif
204
205 // Diffuse
206#if DO_DIFFUSE_LIGHTING
207
208 diffuse *= light.diffuse * max(dot(normal, lightDirection), 0.0f);
209#endif
210
211 // Specular
212#if DO_SPECULAR_HIGHLIGHTS
213
214 float3 viewDirection = normalise(ray.position - currentPosition);
215 float3 reflectDirection = reflect(-lightDirection, normal);
216 specular *= light.specular * pow(max(dot(viewDirection, reflectDirection), 0.0f), material.shininess);
217#endif
218
219#if DO_EDGE_SHADING
220 shadow = 1 - ((float)(steps) / (float)(MAXIMUM_MARCH_STEPS));
221#endif
222
223 // Soft shadows
224#if DO_SOFT_SHADOWS
225 shadow = calculateSoftShadow(currentPosition, time, light.position);
226
227 // Hard shadows
228#elif DO_HARD_SHADOWS
229 shadow = calculateHardShadow(currentPosition, time, light.position);
230#endif
231
232 return ambient + shadow * (diffuse + specular);
233 }
234 }
235
236 // Apply a glow colour around the geometry
237#if DO_GEOMETRY_GLOW
238
239 if (closestDistanceToGeometry <= SCENE_MAX_GLOW_DISTANCE)
240 {
241 float glow_strength = 1 - closestDistanceToGeometry / SCENE_MAX_GLOW_DISTANCE;
242 return SCENE_BACKGROUND_COLOUR + glow_strength * SCENE_GLOW_COLOUR;
243 }
244
245#endif
246
247 return SCENE_BACKGROUND_COLOUR;
248}

258Ray getCameraRay(const float2 screen_coordinate, const float3 camera_position, const float3 camera_facing, const float aspect_ratio)
259{
260 const float theta = CAMERA_VERTICAL_FOV_DEGREES * PI / 180.0f;
261 const float h = tan(theta / 2);
262 const float viewport_height = 2.0f * h;
263 const float viewport_width = aspect_ratio * viewport_height;
264
265 float3 w = normalise(camera_facing);
266 float3 u = normalise(crossProduct(CAMERA_UP_AXIS, w));
267 float3 v = crossProduct(w, u);
268
269 float3 horizontal = u * CAMERA_FOCUS_DISTANCE * viewport_width;
270 float3 vertical = v * CAMERA_FOCUS_DISTANCE * viewport_height;
271 float3 screenLowerLeftCorner = camera_position - horizontal / 2 - vertical / 2 - w * CAMERA_FOCUS_DISTANCE;
272
273 float3 screenPosition = screenLowerLeftCorner + horizontal * screen_coordinate.x + vertical * screen_coordinate.y - camera_position;
274
275 Ray r;
276 r.position = camera_position + screenPosition;
277 r.direction = normalise(screenPosition);
278 return r;
279}
