258 size_t Renderer::calculateWorkGroupSize(size_t total_work_items, size_t maximum_work_group_size, size_t desired_work_group_size)
259 {
260 // User has specified a work group size
261 if (desired_work_group_size > 0)
262 {
263 if (desired_work_group_size > maximum_work_group_size)
264 {
265 log << "Error: Desired work group size (" << desired_work_group_size << ") must be smaller or equal to the maximum work group size (" << total_work_items << ")\n";
266 exit(1);
267 }
268
269 if (total_work_items % desired_work_group_size != 0)
270 {
271 log << "Error: Desired work group size (" << desired_work_group_size << ") must be a multiple of the total work items (total number of pixels " << total_work_items << ")\n";
272 exit(1);
273 }
274
275 return desired_work_group_size;
276 }
277
278 else
279 {
280 // Use default work group size
281 if (total_work_items % maximum_work_group_size == 0)
282 {
283 return maximum_work_group_size;
284 }
285 // Calculate work group size
286 else
287 {
288 for (size_t chosen = maximum_work_group_size; chosen > 1; chosen--)
289 {
290 if (total_work_items % chosen == 0)
291 {
292 return chosen;
293 }
294 }
295
296 return 1;
297 }
298 }
