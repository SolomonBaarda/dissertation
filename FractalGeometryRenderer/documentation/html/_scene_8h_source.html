51 std::tuple<T, T, float> get_camera_value_at_time(const std::vector <std::pair<T, float>>& vector, float time, bool do_loop)
52 {
53 if (vector.size() == 1)
54 {
55 return std::tuple<T, T, float>(vector.at(0).first, vector.at(0).first, vector.at(0).second);
56 }
57 else
58 {
59 // Get the time relative to 0 and the max for this camera path
60 float time_relative = do_loop ? fmod(time, vector.back().second) : time;
61
62 // Before the first element
63 if (time_relative < vector.front().second)
64 {
65 return std::tuple<T, T, float>(vector.front().first, vector.front().first, vector.front().second);
66 }
67 // After the last element
68 else if (time_relative > vector.back().second)
69 {
70 return std::tuple<T, T, float>(vector.back().first, vector.back().first, vector.back().second);
71 }
72 // Otherwise, we must be somewhere between
73 else
74 {
75 for (uint32_t i = 0; i < vector.size() - 1; i++)
76 {
77 const auto& before = vector.at(i);
78 const auto& after = vector.at(i + 1);
79
80 // Current time is somewhere between these two values
81 if (before.second <= time_relative && after.second > time_relative)
82 {
83 return std::tuple<T, T, float>(before.first, after.first, (time_relative - before.second) / (after.second - before.second));
84 }
85 }
86 }
87 }
